import unittest
from tkinter import Tk
from tkinter import Entry
from tkinter import Label
from tkinter import Button
import color_game  # Assuming the provided code is saved in a file named color_game.py

class TestColorGame(unittest.TestCase):

    def setUp(self):
        self.root = Tk()
        self.game = color_game

    def test_startGame(self):
        self.game.startGame()
        self.assertTrue(self.game.timer < 60)  # Check if the timer started

    def test_resetGame(self):
        self.game.resetGame()
        self.assertEqual(self.game.timer, 60)  # Check if the timer reset to 60
        self.assertEqual(self.game.score, 0)    # Check if the score reset to 0

    def test_displayNextWord(self):
        self.game.displayNextWord("<Return>")
        self.assertTrue(self.game.score >= 0)   # Check if score is updated
        self.assertTrue(self.game.timer >= 0)   # Check if timer is updated

    def tearDown(self):
        self.root.destroy()

if __name__ == '__main__':
    unittest.main()
